```python
"""Claude Code Router配置验证脚本"""

import os
import json
import requests
from pathlib import Path

def validate_configuration():
    """验证配置完整性"""
    config_path = Path.home() / ".claude-code-router" / "config.json"
    
    # 检查配置文件存在
    if not config_path.exists():
        return {"status": "error", "message": "配置文件不存在"}
    
    # 读取配置
    with open(config_path, 'r') as f:
        config = json.load(f)
    
    # 验证关键配置
    validation_results = []
    
    # 验证API密钥环境变量
    api_key = os.getenv('DEEPSEEK_API_KEY')
    if not api_key or api_key.startswith('$'):
        validation_results.append({"检查项": "DEEPSEEK_API_KEY", "状态": "失败", "问题": "环境变量未设置"})
    else:
        validation_results.append({"检查项": "DEEPSEEK_API_KEY", "状态": "通过", "详情": "环境变量已设置"})
    
    # 验证提供商配置
    providers = config.get('Providers', [])
    if not providers:
        validation_results.append({"检查项": "提供商配置", "状态": "失败", "问题": "无提供商配置"})
    else:
        validation_results.append({"检查项": "提供商配置", "状态": "通过", "详情": f"配置了{len(providers)}个提供商"})
    
    # 验证路由配置
    router_config = config.get('Router', {})
    if not router_config:
        validation_results.append({"检查项": "路由配置", "状态": "失败", "问题": "无路由配置"})
    else:
        validation_results.append({"检查项": "路由配置", "状态": "通过", "详情": "路由配置完整"})
```

```bash
# 配置参数
SERVICE_NAME="claude-code-router"
HEALTH_CHECK_URL="http://127.0.0.1:3000/health"
MAX_RETRIES=3
RETRY_DELAY=10

check_service_health() {
    if curl -s "$HEALTH_CHECK_URL" > /dev/null; then
        return 0
    else
        return 1
    fi
}

start_service() {
    echo "启动 $SERVICE_NAME..."
    ccr start
    
    # 等待服务启动
    sleep 10
    
    # 检查服务状态
    if check_service_health; then
        echo "✅ $SERVICE_NAME 启动成功"
        return 0
    else
        echo "❌ $SERVICE_NAME 启动失败"
        return 1
    fi
}

# 主循环
while true; do
    if ! check_service_health; then
        echo "⚠️ $SERVICE_NAME 服务异常，尝试重启..."
        
        for attempt in $(seq 1 $MAX_RETRIES); do
            echo "尝试 $attempt/$MAX_RETRIES..."
            
            if start_service; then
                break
            fi
            
            if [ $attempt -eq $MAX_RETRIES ]; then
                echo "❌ 重启失败，达到最大重试次数"
                # 发送告警
                echo "发送告警通知..."
                break
            fi
            
            sleep $RETRY_DELAY
        done
    else
        echo "✅ $SERVICE_NAME 运行正常"
    fi
    
    # 每5分钟检查一次
    sleep 300
Done
```

```python
"""系统进程监控脚本"""

import psutil
import time
import json
from datetime import datetime
from pathlib import Path

class ProcessMonitor:
    """进程监控器"""
    
    def __init__(self):
        self.critical_processes = [
            {
                "name": "claude-code-router",
                "expected_cmd": "claude-code-router",
                "health_check": "http://127.0.0.1:3000/health",
                "restart_command": "ccr start",
                "max_restart_attempts": 3
            },
            {
                "name": "athena-ai-plan-runner", 
                "expected_cmd": "athena_ai_plan_runner",
                "health_check": "queue_status_check",
                "restart_command": "启动队列运行器命令",
                "max_restart_attempts": 3
            },
            {
                "name": "athena-web-desktop",
                "expected_cmd": "athena_web_desktop",
                "health_check": "web_interface_check",
                "restart_command": "启动Web桌面命令",
                "max_restart_attempts": 3
            }
        ]
        
        self.monitor_log = Path("/Volumes/1TB-M2/openclaw/logs/process_monitor.log")
        self.monitor_log.parent.mkdir(parents=True, exist_ok=True)
    
    def find_process_by_command(self, expected_cmd):
        """根据命令查找进程"""
        for proc in psutil.process_iter(['pid', 'name', 'cmdline']):
            try:
                cmdline = ' '.join(proc.info['cmdline'] or [])
                if expected_cmd in cmdline:
                    return proc.info
            except (psutil.NoSuchProcess, psutil.AccessDenied):
                continue
        return None
    
    def check_process_health(self, process_config):
        """检查进程健康状态"""
        process_info = self.find_process_by_command(process_config["expected_cmd"])
        
        if not process_info:
            return {
                "status": "not_found",
                "message": f"进程 {process_config['name']} 未找到"
            }
        
        # 检查进程是否响应
        try:
            if process_config["health_check"] == "http://127.0.0.1:3000/health":
                import requests
                response = requests.get(process_config["health_check"], timeout=5)
                if response.status_code == 200:
                    return {
                        "status": "healthy",
                        "pid": process_info["pid"],
                        "message": "进程运行正常"
                    }
                else:
                    return {
                        "status": "unhealthy",
                        "pid": process_info["pid"],
                        "message": f"健康检查失败: {response.status_code}"
                    }
            else:
                # 其他健康检查逻辑
                return {
                    "status": "healthy",
                    "pid": process_info["pid"],
                    "message": "进程运行正常"
                }
        except Exception as e:
            return {
                "status": "error",
                "pid": process_info["pid"],
                "message": f"健康检查异常: {str(e)}"
            }
    
    def monitor_all_processes(self):
        """监控所有关键进程"""
        results = {}
        
        for process_config in self.critical_processes:
            health_status = self.check_process_health(process_config)
            results[process_config["name"]] = health_status
            
            # 记录日志
            self.log_status(process_config["name"], health_status)
            
            # 触发告警
            if health_status["status"] != "healthy":
                self.trigger_alert(process_config["name"], health_status)
        
        return results
    
    def log_status(self, process_name, health_status):
        """记录状态日志"""
        log_entry = {
            "timestamp": datetime.now().isoformat(),
            "process": process_name,
            "status": health_status["status"],
            "message": health_status["message"]
        }
        
        with open(self.monitor_log, 'a') as f:
            f.write(json.dumps(log_entry) + '\n')
    
    def trigger_alert(self, process_name, health_status):
        """触发告警"""
        alert_message = f"⚠️ 进程告警: {process_name} - {health_status['message']}"
        print(alert_message)
        
        # 这里可以集成邮件、Slack等告警渠道
        # self.send_slack_alert(alert_message)
        # self.send_email_alert(alert_message)

# 使用示例
if __name__ == "__main__":

    while True:
        print("🔍 检查系统进程状态...")
        results = monitor.monitor_all_processes()
        print(json.dumps(results, indent=2, ensure_ascii=False))
        
        # 每30秒检查一次
```

```python
# 性能数据分析工具: performance_analyzer.py
"""性能数据分析工具"""

import json
import pandas as pd
import matplotlib.pyplot as plt
from datetime import datetime
from pathlib import Path

class PerformanceAnalyzer:
    """性能数据分析器"""
    
    def __init__(self, data_dir):
        self.data_dir = Path(data_dir)
        self.df = None
    
    def load_performance_data(self):
        """加载性能数据"""
        metrics_file = self.data_dir / "performance" / "performance_metrics.jsonl"
        
        if not metrics_file.exists():
            raise FileNotFoundError(f"性能指标文件不存在: {metrics_file}")
        
        # 读取JSONL文件
        data = []
        with open(metrics_file, 'r') as f:
            for line in f:
                if line.strip():
                    data.append(json.loads(line.strip()))
        
        self.df = pd.DataFrame(data)
        return self.df
    
    def analyze_cpu_trends(self):
        """分析CPU使用趋势"""
        if self.df is None:
            self.load_performance_data()
        
        # 提取CPU相关指标
        cpu_data = self.df[['timestamp', 'cpu.usage_percent', 'cpu.load_average']].copy()
        cpu_data['timestamp'] = pd.to_datetime(cpu_data['timestamp'])
        
        # 计算统计指标
        cpu_stats = {
            "平均使用率": cpu_data['cpu.usage_percent'].mean(),
            "峰值使用率": cpu_data['cpu.usage_percent'].max(),
            "使用率标准差": cpu_data['cpu.usage_percent'].std(),
            "稳定性评估": "优秀" if cpu_data['cpu.usage_percent'].std() < 10 else "需要优化"
        }
        
        return cpu_stats
    
    def identify_bottlenecks(self):
        """识别性能瓶颈"""
        if self.df is None:
            self.load_performance_data()
        
        bottlenecks = []
        
        # CPU瓶颈识别
        high_cpu_periods = self.df[self.df['cpu.usage_percent'] > 80]
        if len(high_cpu_periods) > 0:
            bottlenecks.append({
                "类型": "CPU瓶颈",
                "严重程度": "高",
                "描述": f"检测到{len(high_cpu_periods)}次CPU使用率超过80%",
                "建议": "优化计算密集型任务，考虑负载均衡"
            })
        
        # 内存瓶颈识别
        low_memory_periods = self.df[self.df['memory.pressure_free_percent'] < 20]
        if len(low_memory_periods) > 0:
            bottlenecks.append({
                "类型": "内存瓶颈", 
                "严重程度": "高",
                "描述": f"检测到{len(low_memory_periods)}次空闲内存低于20%",
                "建议": "优化内存使用，考虑增加内存或优化内存管理"
            })
        
        return bottlenecks
    
    def generate_report(self):
        """生成分析报告"""
        cpu_stats = self.analyze_cpu_trends()
        bottlenecks = self.identify_bottlenecks()
        
        report = {
            "分析时间": datetime.now().isoformat(),
            "数据总量": len(self.df),
            "时间范围": {
                "开始": self.df['timestamp'].min(),
                "结束": self.df['timestamp'].max()
            },
            "CPU性能分析": cpu_stats,
            "性能瓶颈": bottlenecks,
            "优化建议": self.generate_optimization_suggestions(bottlenecks)
        }
        
        return report
    
    def generate_optimization_suggestions(self, bottlenecks):
        """生成优化建议"""
        suggestions = []
        
        for bottleneck in bottlenecks:
            if bottleneck["类型"] == "CPU瓶颈":
                suggestions.append({
                    "优先级": "高",
                    "建议": "实现任务队列和负载均衡",
                    "预期效果": "降低CPU峰值使用率20-30%"
                })
            elif bottleneck["类型"] == "内存瓶颈":
                suggestions.append({
                    "优先级": "高", 
                    "建议": "优化内存分配和垃圾回收策略",
                    "预期效果": "减少内存压力15-25%"
                })
        
        return suggestions

# 使用示例

    report = analyzer.generate_report()
```

```python
# Agent异常处理框架: agent_exception_handler.py
"""Agent异常处理框架"""

import logging
import traceback
from datetime import datetime
from typing import Dict, Any, Callable
from enum import Enum

class ExceptionSeverity(Enum):
    """异常严重程度"""
    LOW = "low"
    MEDIUM = "medium" 
    HIGH = "high"
    CRITICAL = "critical"

class AgentExceptionHandler:
    """Agent异常处理器"""
    
    def __init__(self, agent_name: str):
        self.agent_name = agent_name
        self.logger = logging.getLogger(f"agent.{agent_name}")
        
        # 异常处理策略
        self.recovery_strategies = {
            "network_error": self._handle_network_error,
            "timeout_error": self._handle_timeout_error,
            "validation_error": self._handle_validation_error,
            "resource_error": self._handle_resource_error
        }
        
        # 异常统计
        self.exception_stats = {
            "total_count": 0,
            "by_type": {},
            "recovery_success": 0,
            "recovery_failure": 0
        }
    
    def handle_exception(self, exception: Exception, context: Dict[str, Any] = None) -> Dict[str, Any]:
        """处理异常"""
        self.exception_stats["total_count"] += 1
        
        # 分析异常类型
        exception_type = type(exception).__name__
        severity = self._assess_severity(exception, context)
        
        # 更新统计
        if exception_type not in self.exception_stats["by_type"]:
            self.exception_stats["by_type"][exception_type] = 0
        self.exception_stats["by_type"][exception_type] += 1
        
        # 记录异常
        self._log_exception(exception, severity, context)
        
        # 尝试恢复
        recovery_result = self._attempt_recovery(exception, context)
        
        return {
            "agent": self.agent_name,
            "exception_type": exception_type,
            "severity": severity.value,
            "recovery_attempted": recovery_result["attempted"],
            "recovery_successful": recovery_result["successful"],
            "recovery_strategy": recovery_result["strategy"],
            "suggested_action": recovery_result["suggested_action"],
            "timestamp": datetime.now().isoformat()
        }
    
    def _assess_severity(self, exception: Exception, context: Dict[str, Any]) -> ExceptionSeverity:
        """评估异常严重程度"""
        exception_type = type(exception).__name__
        
        severity_mapping = {
            "ConnectionError": ExceptionSeverity.HIGH,
            "TimeoutError": ExceptionSeverity.MEDIUM,
            "ValidationError": ExceptionSeverity.LOW,
            "MemoryError": ExceptionSeverity.CRITICAL,
            "KeyError": ExceptionSeverity.MEDIUM
        }
        
        return severity_mapping.get(exception_type, ExceptionSeverity.MEDIUM)
    
    def _attempt_recovery(self, exception: Exception, context: Dict[str, Any]) -> Dict[str, Any]:
        """尝试恢复"""
        exception_type = type(exception).__name__
        
        # 根据异常类型选择恢复策略
        recovery_strategy = None
        for pattern, strategy in self.recovery_strategies.items():
            if pattern in exception_type.lower():
                recovery_strategy = strategy
                break
        
        if recovery_strategy:
            try:
                result = recovery_strategy(exception, context)
                self.exception_stats["recovery_success"] += 1
                return {
                    "attempted": True,
                    "successful": True,
                    "strategy": recovery_strategy.__name__,
                    "suggested_action": result.get("suggested_action", "继续执行")
                }
            except Exception as recovery_error:
                self.exception_stats["recovery_failure"] += 1
                self._log_exception(recovery_error, ExceptionSeverity.MEDIUM, {"recovery_attempt": True})
                return {
                    "attempted": True,
                    "successful": False,
                    "strategy": recovery_strategy.__name__,
                    "suggested_action": "需要人工干预"
                }
        else:
            return {
                "attempted": False,
                "successful": False,
                "strategy": "none",
                "suggested_action": "无自动恢复策略"
            }
    
    def _handle_network_error(self, exception: Exception, context: Dict[str, Any]) -> Dict[str, Any]:
        """处理网络错误"""
        # 实现网络错误恢复逻辑
        # 例如：重试、切换备用服务等
        return {
            "suggested_action": "重试操作或切换到备用服务",
            "retry_count": 3,
            "backoff_delay": 5
        }
    
    def _handle_timeout_error(self, exception: Exception, context: Dict[str, Any]) -> Dict[str, Any]:
        """处理超时错误"""
        # 实现超时错误恢复逻辑
        return {
            "suggested_action": "增加超时时间或优化处理逻辑",
            "timeout_increase": "+30秒"
        }
    
    def _log_exception(self, exception: Exception, severity: ExceptionSeverity, context: Dict[str, Any]):
        """记录异常日志"""
        log_message = f"[{severity.value}] {self.agent_name} 异常: {str(exception)}"
        
        if severity == ExceptionSeverity.CRITICAL:
            self.logger.error(log_message, extra={"context": context})
        elif severity == ExceptionSeverity.HIGH:
            self.logger.warning(log_message, extra={"context": context})
        else:
            self.logger.info(log_message, extra={"context": context})
        
        # 记录堆栈跟踪（仅在严重异常时）
        if severity in [ExceptionSeverity.HIGH, ExceptionSeverity.CRITICAL]:
            self.logger.debug(f"堆栈跟踪: {traceback.format_exc()}")
    
    def get_exception_statistics(self) -> Dict[str, Any]:
        """获取异常统计"""
        return {
            "agent": self.agent_name,
            "statistics": self.exception_stats,
            "recovery_rate": (
                self.exception_stats["recovery_success"] / 
                max(1, self.exception_stats["recovery_success"] + self.exception_stats["recovery_failure"])
            ) * 100
        }

# 使用示例
if __name__ == "__main__":
    # 配置日志
    logging.basicConfig(level=logging.INFO)
    
    # 创建异常处理器
    handler = AgentExceptionHandler("research_agent")
    
    # 模拟异常处理
    try:
        # 模拟一个网络错误
        raise ConnectionError("连接被拒绝")
    except Exception as e:
        result = handler.handle_exception(e, {"operation": "数据获取", "target": "api.example.com"})
        print("异常处理结果:", json.dumps(result, indent=2, ensure_ascii=False))
    
    # 查看统计
    stats = handler.get_exception_statistics()
    print("异常统计:", json.dumps(stats, indent=2, ensure_ascii=False))
```
