### 5.2 关键算法实现

#### 分布稳定性计算算法
```python
import numpy as np
from sklearn.cluster import HDBSCAN
from sklearn.metrics import silhouette_score
import sentence_transformers

class DistributionStabilityAnalyzer:
    """分布稳定性分析器"""
    
    def __init__(self):
        self.model = sentence_transformers.SentenceTransformer('all-MiniLM-L6-v2')
    
    def calculate_stability(self, outputs: list) -> float:
        """
        计算输出分布稳定性
        Args:
            outputs: 相同输入的多次输出结果列表
        Returns:
            稳定性得分 (0-1)，1表示完全稳定
        """
        if len(outputs) < 10:
            return 0.0  # 样本不足
        
        # 将文本输出转换为语义向量
        embeddings = self.model.encode(outputs)
        
        # 使用HDBSCAN进行聚类
        clusterer = HDBSCAN(min_samples=5, min_cluster_size=2)
        cluster_labels = clusterer.fit_predict(embeddings)
        
        # 计算聚类质量
        n_clusters = len(set(cluster_labels)) - (1 if -1 in cluster_labels else 0)
        
        if n_clusters > 1:
            # 多簇情况，计算轮廓系数
            stability_score = silhouette_score(embeddings, cluster_labels)
            # 归一化到0-1范围
            stability_score = (stability_score + 1) / 2
        else:
            # 单一簇，视为高度稳定
            stability_score = 1.0
        
        return stability_score
    
    def detect_drift(self, historical_outputs: list, current_outputs: list) -> float:
        """
        检测分布漂移
        Args:
            historical_outputs: 历史输出样本
            current_outputs: 当前输出样本
        Returns:
            漂移程度 (0-1)，0表示无漂移
        """
        hist_embeddings = self.model.encode(historical_outputs)
        curr_embeddings = self.model.encode(current_outputs)
        
        # 计算两个分布的中心距离
        hist_center = np.mean(hist_embeddings, axis=0)
        curr_center = np.mean(curr_embeddings, axis=0)
        
        # 使用余弦相似度计算漂移程度
        similarity = np.dot(hist_center, curr_center) / (
            np.linalg.norm(hist_center) * np.linalg.norm(curr_center)
        )
        
        # 转换为漂移程度 (1 - 相似度)
        drift_score = 1 - similarity
        
        return drift_score
```

#### 自适应负载控制算法
```python
class AdaptiveLoadController:
    """自适应负载控制器"""
    
    def __init__(self):
        self.health_thresholds = {
            'latency_p99': 5.0,  # 秒
            'error_rate': 0.01,  # 1%
            'cpu_usage': 0.8,    # 80%
            'memory_usage': 0.8  # 80%
        }
    
    def calculate_health_score(self, metrics: dict) -> float:
        """计算系统健康度得分"""
        # 归一化各项指标
        latency_score = min(metrics['latency_p99'] / self.health_thresholds['latency_p99'], 1.0)
        error_score = min(metrics['error_rate'] / self.health_thresholds['error_rate'], 1.0)
        cpu_score = min(metrics['cpu_usage'] / self.health_thresholds['cpu_usage'], 1.0)
        memory_score = min(metrics['memory_usage'] / self.health_thresholds['memory_usage'], 1.0)
        
        # 加权计算综合健康度
        weights = {'latency': 0.3, 'error': 0.3, 'cpu': 0.2, 'memory': 0.2}
        health_score = (
            weights['latency'] * latency_score +
            weights['error'] * error_score +
            weights['cpu'] * cpu_score +
            weights['memory'] * memory_score
        )
        
        return health_score
    
    def adjust_load(self, current_metrics: dict, current_load: int) -> dict:
        """基于健康度调整负载"""
        health_score = self.calculate_health_score(current_metrics)
        
        if health_score > 0.8:
            # 系统不健康，降低负载
            new_load = int(current_load * 0.7)
            action = "decrease"
        elif health_score < 0.3:
            # 系统健康，增加负载
            new_load = int(current_load * 1.3)
            action = "increase"
        else:
            # 系统状态良好，保持当前负载
            new_load = current_load
            action = "maintain"
        
        return {
            'action': action,
            'new_load': new_load,
            'health_score': health_score,
            'timestamp': time.time()
        }
```
